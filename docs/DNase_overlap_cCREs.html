#-----------------------------
# Helper: BED (0-based start, end exclusive or inclusive doesn't matter much if consistent)
# We convert to 1-based inclusive GRanges: start = start0+1, end = end
#-----------------------------
bed_to_gr <- function(dt, keep_chr = NULL) {
  stopifnot(ncol(dt) >= 3)
  dt <- copy(dt)
  setnames(dt, 1:3, c("chr","start0","end"))
  gr <- GRanges(
    seqnames = dt$chr,
    ranges   = IRanges(start = dt$start0 + 1L, end = dt$end)
  )
  if (!is.null(keep_chr)) gr <- gr[seqnames(gr) %in% keep_chr]
  gr
}
